    }

    @Override

        sb.append("<ul>\n");

        }
        sb.append("</ul>/n/n");
    }

    @Override
    public void close() {
        sb.append("</body></html>\n");
    }

    public String getHTMLResult() {
        return sb.toString();
    }
}
